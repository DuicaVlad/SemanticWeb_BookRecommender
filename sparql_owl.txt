# Query 1: List all books and their authors
PREFIX ex: <http://example.org/bookstore/ontology#>
SELECT ?book ?title ?author
WHERE {
  ?book a ex:Book ;
        ex:hasTitle ?title ;
        ex:hasAuthor ?author .
}

# Query 2: List all Users and their preferred themes
PREFIX ex: <http://example.org/bookstore/ontology#>
SELECT ?user ?name ?theme
WHERE {
  ?user a ex:User ;
        ex:hasName ?name ;
        ex:prefersTheme ?theme .
}

# Query 3: Find all books suitable for "Beginner" level
PREFIX ex: <http://example.org/bookstore/ontology#>
SELECT ?title
WHERE {
  ?book a ex:Book ;
        ex:hasTitle ?title ;
        ex:suitableForLevel "Beginner" .
}

# Query 4: Find all books with the theme "Science Fiction"
PREFIX ex: <http://example.org/bookstore/ontology#>
SELECT ?title ?author
WHERE {
  ?book a ex:Book ;
        ex:hasTitle ?title ;
        ex:hasAuthor ?author ;
        ex:hasTheme "Science Fiction" .
}
